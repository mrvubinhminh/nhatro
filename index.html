const SHEET_NAME = "Database"; // Hàm xử lý khi app gửi dữ liệu lên (Lưu dữ liệu) function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === 'saveRooms') { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); // Nếu chưa có sheet tên "Database", tự động tạo mới if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(["RoomID", "RoomDataJSON"]); } const rooms = data.rooms; // Xoá dữ liệu cũ (chừa lại dòng tiêu đề) if (sheet.getLastRow() > 1) { sheet.getRange(2, 1, sheet.getLastRow() - 1, 2).clearContent(); } // Chuẩn bị dữ liệu mới để ghi // Cột A: ID Phòng | Cột B: Toàn bộ dữ liệu của phòng đó const rows = rooms.map(room => [room.id, JSON.stringify(room)]); // Ghi hàng loạt vào Google Sheets (rất nhanh) if (rows.length > 0) { sheet.getRange(2, 1, rows.length, 2).setValues(rows); } return ContentService.createTextOutput(JSON.stringify({ success: true })) .setMimeType(ContentService.MimeType.JSON); } } catch (err) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // Hàm xử lý khi app lấy dữ liệu về (Tải dữ liệu) function doGet(e) { try { if (e.parameter.action === 'getRooms') { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); // Nếu chưa có dữ liệu, trả về mảng rỗng if (!sheet || sheet.getLastRow() <= 1) { return ContentService.createTextOutput(JSON.stringify([])) .setMimeType(ContentService.MimeType.JSON); } // Lấy toàn bộ dữ liệu ở cột B const dataRange = sheet.getRange(2, 1, sheet.getLastRow() - 1, 2).getValues(); const rooms = dataRange.map(row => { try { return JSON.parse(row[1]); } catch (error) { return null; } }).filter(r => r !== null); return ContentService.createTextOutput(JSON.stringify(rooms)) .setMimeType(ContentService.MimeType.JSON); } } catch (err) { return ContentService.createTextOutput(JSON.stringify({ error: err.message })) .setMimeType(ContentService.MimeType.JSON); } }
